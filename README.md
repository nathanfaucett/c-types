types
=======

better types for c

byte - 8 bit;
bool - 8 bit;

intsize - 32 or 64 bit;
int8 - 8 bit;
int16 - 16 bit;
int32 - 32 bit;
int64 - 64 bit;

uintsize - 32 or 64 bit;
uint8 - 8 bit;
uint16 - 16 bit;
uint32 - 32 bit;
uint64 - 64 bit;

float32 - 32 bit;
float64 - 64 bit;


true - 1;
false - 0;
